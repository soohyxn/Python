import cv2 as cv

cap = cv.VideoCapture('stopwatch.avi')

if not cap.isOpened():
    print("video open failed")
    exit()

w = round(cap.get(cv.CAP_PROP_FRAME_WIDTH))
h = round(cap.get(cv.CAP_PROP_FRAME_HEIGHT))
fps = cap.get(cv.CAP_PROP_FPS)

fourcc = cv.VideoWriter_fourcc(*'DIVX')
delay = round(1000/fps)
outputVideo = cv.VideoWriter('outputB.avi', fourcc, fps, (w,h))

if not outputVideo.isOpened():
    print('file open failed')
    exit()

while True:
    ret, frame = cap.read()
    if not ret:
        break

    inversed = frame.copy()
    if cap.get(cv.CAP_PROP_POS_MSEC) >= 9000:
        inversed[240:, 320:] = ~inversed[240:, 320:]
    outputVideo.write(inversed)

    cv.imshow('frame', frame)
    cv.imshow('inversed', inversed)

    if cv.waitKey(delay) == 27:
        break

    cv.destroyAllWindows()
